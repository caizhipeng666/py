### 协程生成器
```python
import inspect


def iter_test(first):
    print("first:", first)
    second = yield first
    print("second:", second)
    third = yield first + second
    print("third:", third)
    return third


try:
    test = iter_test(1)
    print(inspect.getgeneratorstate(test))
    current = next(test)
    print("current:", current)
    current = test.send(3)
    print("current:", current)
    print(inspect.getgeneratorstate(test))
    test.send(3)
except StopIteration:
    print("stop")

>>> GEN_CREATED
    first: 1
    current: 1
    second: 3
    current: 4
    GEN_SUSPENDED
    third: 3
    stop
```
